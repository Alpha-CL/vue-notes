#### getMountDomOfComp usage

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * getComponentRootDom(comp, props): $el;
 * 
 * 
 * 获取组件的真实挂载节点
 */


function getComponentRootDom(comp, props) {

    const vm = new Vue({
        render: (h) => h(comp, {props})
    });

    vm.$mount();

    return vm.$el;
}


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```
