## throw event

#### $emit(eventName: string, $event, ...params);

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * 子组件 抛出事件并传递参数，父组件 决定该如何改变数据           // 当数据由父组件提供时
 * 
 * 
 * 因 vue是单向数据流，谁的数据由谁控制，保证数据的可维护性
 */


this.$emit(eventName: string, $event, ...params);


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


                    FatherComp: 数据属于父组件的情况
                    
                    1. 利用 props 属性，传输数据到 SonComp
                    2. 处理 SonComp 抛出的事件及事件参数
                    
                    FatherComp: {
                    
                        methods: {
                        
                            sonEvent(e, ...params){
                            
                                ...
                            }
                        }
                    }

                    + ------------------------- +
                    |                           |
                    |        Father Comp        |
            + <---- +                           | <---- +
            |       |     data, handle-even     |       |
            |       |                           |       |
            |       + ------------------------- +       |
            |                                           |
       transfer-data                               throw-event 
            |                                           |
            |       + ------------------------- +       |
            |       |                           |       |
            |       |         Son Comp          |       |
            + ----> |                           | ----> +
                    |     props, throw-event    |
                    |                           |
                    + ------------------------- +

                    SonComp: 谁的数据谁处理，保证单向数据流
                     
                    1. 利用 FatherComp 传输的数据创建组件
                    2. 将需要 FatherComp 的数据处理通过 $emit(); 抛出

                    SonComp: {
                    
                        methods: {
                                            
                            customEvent(e, ...params){
                            
                                this.$emit('sonEvent', ...params);
                            }
                        }
                    }


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

#### default slot

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * 默认插槽
 * 
 * 
 * 
 */

//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


// FatherComp

<template>
    <Son>
        <!-- default-slot -->
    </Son>
</template>


// SonComp

<template>
    <slot/>
</template>


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

#### named slot

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * 具名插槽
 * 
 * 
 * 
 */

<template v-slot:customSlot/> === <template #customSlot/>


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


// FatherComp

<template>
    <Son>
    
        <!-- default-slot -->           // 未在具名插槽内的自元素将作为 默认插槽的内容传递
        
        <template v-slot:namedSlot>
                <!-- any -->
        </template>
        
    </Son>
</template>


// SonComp

<template>
    <slot/>                             // <slot name="default"/>
    <slot name="namedSlot"/>
</template>


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```
