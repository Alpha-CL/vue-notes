#### get mount-dom of component

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * getMountDomOfComp(comp, props): $el;
 * 
 * 
 * 获取组件的真实挂载节点
 */


function getMountDomOfComp(comp, props) {

    const vm = new Vue({
        render: (h) => h(comp, {props})
    });

    vm.$mount();

    return vm.$el;
}


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

#### extend vue member

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * 自定义 Vue.prototy 成员 
 * 
 * 
 ** 需对自定义扩展成员特殊命名，避免耦合 eg: __[myMember]
 */


Vue.prototype.[myMember] = any;                       // 扩展 vue 实例

const vm = new Vue({}).mount('#app');


vm.__[myMember];                    // 实例 可调用扩展的成员

Comp: {                         

    method: {
    
        handle() {
        
            this.__[myMember]       // 组件中通过 this 可调用扩展成员
        }
    }
}


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```
